	A = 1, B = 2, C = 3, X = 20, Y = 10, Z = 1;
	V1 = True, V2 = False;
	Nome = 'Pedro', Rua = 'Pedrinho'.
	
	Encontre o resultado das expressões lógicas abaixo:
	a) A + C / B
	b) C / B / A
	C) -X ** B
	d) (-X) ** B
	e) V1 or V2
	f) V1 and not V2
	g) V2 and not V1
	h) not Nome == Rua
	i) V1 and not V2 or V2 and not True
	j) X > Y and C <= B
	k) C - 3 * A < X + 2 * Z
